535  {
536  Socket socket = null;
537  Debug.Print("connectSocket started");
538  try
539  {
540  lock (toSend)
541  {
542  toSend.Clear();
543  }
544  Debug.Print("Create new Socket..");
545  socket = new Socket(AddressFamily.InterNetwork, SocketType.Stream, ProtocolType.Tcp);
546  String ipAddr = "192.168.137.1";
547 
548  Debug.Print("Generate EndPoint..");
549  IPEndPoint remoteEP = new IPEndPoint(IPAddress.Parse(ipAddr), 13000);
550  Debug.Print("Connect..");
551  socket.Connect(remoteEP);
552 
553  socket.SendTimeout = 10000;
554  camera.StartStreamingBitmaps(new Bitmap(camera.CurrentPictureResolution.Width, camera.CurrentPictureResolution.Height));
555  while (true)
556  {
557  bool noBitmap = true;
558  while (noBitmap)
559  {
560  lock (toSend)
561  {
562  if (toSend.Count != 0)
563  noBitmap = false;
564  }
565  if (noBitmap)
566  evento.WaitOne();
567  }
568  MyBitmap myBmp;
569  lock (toSend)

576  currentBitmapData = myBmp.bitmap;

578  Util.BitmapToBMPFile(currentBitmapData.GetBitmap(), currentBitmapData.Width, currentBitmapData.Height, outputFileBuffer);
579  socket.Send(outputFileBuffer);

581  else
582  break;
583 
584  }

587  }
